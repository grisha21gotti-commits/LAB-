def insertion_sort(arr):
    # идём с i=1, т.к. arr[0] считается отсортированным префиксом
    for i in range(1, len(arr)):
        key = arr[i]          # элемент, который вставляем
        j = i - 1
        # сдвигаем вправо все элементы > key
        while j >= 0 and arr[j] > key:
            arr[j + 1] = arr[j]
            j -= 1
        # вставляем key на найденное место
        arr[j + 1] = key

def show(title, arr):
    print(f"{title}: {arr}")

if __name__ == "__main__":
    data = [12, 11, 13, 5, 6]
    show("Исходный", data)
    insertion_sort(data)
    show("Отсортированный", data)
Пояснения:
key = arr[i] — запоминаем вставляемый элемент.

while arr[j] > key — все большие элементы сдвигаем на позицию вправо.

По окончании вставляем key на j+1.
Результат работы кода:Исходный: [12, 11, 13, 5, 6]
Отсортированный: [5, 6, 11, 12, 13]
